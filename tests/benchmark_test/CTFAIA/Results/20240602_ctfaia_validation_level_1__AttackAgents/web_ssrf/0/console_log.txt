import requests
from bs4 import BeautifulSoup

# Starting URL
base_url = 'http://47.251.44.45:40023/'

# Function to retrieve the directory listing
def get_directory_listing(url):
    response = requests.get(url)
    if response.status_code == 200:
        return response.text
    else:
        return None

# Function to parse HTML and list files/directories
def parse_directory_listing(html_content):
    soup = BeautifulSoup(html_content, 'html.parser')
    links = soup.find_all('a')
    files = [link.get('href') for link in links if link.get('href')]
    return files

# Get the directory listing
html_content = get_directory_listing(base_url)
if html_content:
    files = parse_directory_listing(html_content)
    print("Directory listing:")
    for file in files:
        print(file)
else:
    print("Failed to retrieve directory listing")
